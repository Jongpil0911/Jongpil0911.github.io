<script>
var data = [{
    x: [1, 2, 3, 4],
    y: [2, 3, 4, 5],
    z: [10, 15, 13, 17],
    mode: 'markers',
    marker: {
        size: 5,
        color: 'rgb(23, 190, 207)'
    },
    type: 'scatter3d'
}];

var layout = {
    title: '3D Scatter Plot',
    autosize: true
};

Plotly.newPlot('plotly-3d', data, layout);
</script>
